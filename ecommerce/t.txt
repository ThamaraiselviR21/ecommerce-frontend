useEffect(() => {
    const fetchProducts = async () => {
        try {
            const res = await axios.get(`${process.env.REACT_APP_API_URL}/products`, {
                params: searchParams,
            });
            setProducts(res.data.products);
        } catch (error) {
            console.error("Error fetching products:", error);
        }
    };

    fetchProducts();
}, [searchParams]);
